CREATED DATABASE COMA_BEM;

USE COMA_BEM;

CREATE TABLE IF NOT EXISTS `COMA_BEM`.`UNIDADE` (
  `UNI_COD` INT NOT NULL AUTO_INCREMENT,
  `UNI_MEDIDA` VARCHAR(45) NULL,
  PRIMARY KEY (`UNI_COD`))
ENGINE = InnoDB;

CREATE TABLE IF NOT EXISTS `COMA_BEM`.`PRODUTOS` (
  `PRO_COD` INT NOT NULL AUTO_INCREMENT,
  `PROD_NOME` VARCHAR(50) NULL,
  `PROD_VALOR` DOUBLE NULL,
  `PROD_QUANTIDADE` DOUBLE NULL,
  `UNIDADE_UNI_COD` INT NOT NULL,
  PRIMARY KEY (`PRO_COD`),
  INDEX `fk_PRODUTOS_UNIDADE_idx` (`UNIDADE_UNI_COD` ASC) VISIBLE,
  CONSTRAINT `fk_PRODUTOS_UNIDADE`
    FOREIGN KEY (`UNIDADE_UNI_COD`)
    REFERENCES `COMA_BEM`.`UNIDADE` (`UNI_COD`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION)
ENGINE = InnoDB;
